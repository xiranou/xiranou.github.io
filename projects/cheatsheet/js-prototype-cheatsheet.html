JavaScript prototypes

  • Object Constructor

    var person = function(firstName, lastName, age){ this.firstName = firstName; this.lastName = lastName; this.age = age; }

  • Using new keyword to create new objects

    var me = new person('Xiran', 'Ou', 25); var joe = new person('Joseph', 'Ou', 28);

  • Add property to objects

    me.favFood = 'Rice Noodles'; joe.favFood = 'Cake';

  • Add method to objects

    me.sayHello = function(){ return "Hello, my name is " + this.firstName + " " + this.lastName; } joe.sayHello = function(){ return "Wassup!" }

  • Add property to prototypes

    var person = function(firstName, lastName, age){ this.firstName = firstName; this.lastName = lastName; this.age = age; this.fullName = firstName + " " + lastName; }

  • Add method to prototypes

    var person = function(firstName, lastName, age){ this.firstName = firstName; this.lastName = lastName; this.age = age; this.fullName = firstName + " " + lastName; this.sayHello = function(){ return "Hello, my name is " + this.fullName; } }

  • Using prototype Property

    person.prototype.fullName = this.firstName + " " + this.lastName; person.prototype.sayHello = function(){ return "Hello, my name is " + this.fullName; }
